const USERS={
  admin:{pass:'admin123',role:'admin',name:'Admin',email:'admin@factory.io'},

};
let currentUser=null,keycloakToken=null;

function attemptLogin(){
  const u=document.getElementById('login-user').value.trim();
  const p=document.getElementById('login-pass').value;
  const err=document.getElementById('login-error');
  if(!USERS[u]||USERS[u].pass!==p){err.textContent='⚠ Authentication failed. Invalid credentials.';return;}
  err.textContent='';
  currentUser={...USERS[u],username:u};
  keycloakToken={
    access_token:'eyJ'+btoa(JSON.stringify({sub:u,realm_access:{roles:[currentUser.role]},exp:Date.now()+3600000})),
    token_type:'Bearer',expires_in:3600,
    refresh_token:'rt_'+Math.random().toString(36).slice(2),
    scope:'openid profile email',
    session_state:crypto.randomUUID?crypto.randomUUID():Math.random().toString(36),
  };
  sessionStorage.setItem('kc_token',JSON.stringify(keycloakToken));
  sessionStorage.setItem('kc_user',JSON.stringify(currentUser));
  showApp();
}

}

  const ss=keycloakToken?.session_state;

let sqlDB=null,SQL=null;

async function initSQLite(){
  try{
    SQL=await initSqlJs({locateFile:f=>`https://cdnjs.cloudflare.com/ajax/libs/sql.js/1.12.0/${f}`});
    sqlDB=new SQL.Database();
    sqlDB.run(`CREATE TABLE IF NOT EXISTS sensor_readings(id INTEGER PRIMARY KEY AUTOINCREMENT,machine TEXT,sensor TEXT,value REAL,z_score REAL,anomaly_type TEXT DEFAULT 'nominal',risk_score INTEGER,timestamp TEXT)`);
    sqlDB.run(`CREATE TABLE IF NOT EXISTS alerts(id INTEGER PRIMARY KEY AUTOINCREMENT,machine TEXT,type TEXT,severity TEXT,message TEXT,risk INTEGER,timestamp TEXT)`);
    sqlDB.run(`CREATE TABLE IF NOT EXISTS maintenance_schedule(id INTEGER PRIMARY KEY AUTOINCREMENT,machine TEXT,priority INTEGER,description TEXT,scheduled_time TEXT,created_at TEXT)`);
    sqlDB.run(`CREATE TABLE IF NOT EXISTS auth_log(id INTEGER PRIMARY KEY AUTOINCREMENT,username TEXT,action TEXT,timestamp TEXT)`);
    sqlDB.run(`INSERT INTO auth_log(username,action,timestamp)VALUES(?,?,?)`,[currentUser?.username||'?','login',new Date().toISOString()]);
    document.getElementById('db-label').textContent='SQLite ✓';
    logAPI('GET','/db/init','4 tables created','200');
  }catch(e){
    console.error('SQLite init failed:',e);
    document.getElementById('db-dot').className='sb-dot warn';
    document.getElementById('db-label').textContent='SQLite ✗';
  }
}

function dbInsert(machine,sensor,value,z,aType,risk){
  if(!sqlDB)return;
  sqlDB.run(`INSERT INTO sensor_readings(machine,sensor,value,z_score,anomaly_type,risk_score,timestamp)VALUES(?,?,?,?,?,?,?)`,[machine,sensor,value,z,aType,risk,new Date().toISOString()]);
}
function dbInsertAlert(machine,type,sev,msg,risk){
  if(!sqlDB)return;
  sqlDB.run(`INSERT INTO alerts(machine,type,severity,message,risk,timestamp)VALUES(?,?,?,?,?,?)`,[machine,type,sev,msg,risk,new Date().toISOString()]);
}
function dbRowCount(){if(!sqlDB)return 0;const r=sqlDB.exec('SELECT COUNT(*)FROM sensor_readings');return r.length?r[0].values[0][0]:0;}
function exportDB(){
  if(!sqlDB)return;
  const d=sqlDB.export();const b=new Blob([d],{type:'application/octet-stream'});

  logAPI('GET','/db/export','binary','200');
}

// ================= API BASE =================
const BASE_URL = "http://localhost:8000"; // change later


// ================= LOAD MACHINES =================
async function apiLoadMachines(){
  const res = await fetch(BASE_URL + "/machines");
  const data = await res.json();

  data.forEach(m=>{
    state.readings[m.id] = {};
  });

  state.selectedMachine = data[0]?.id;

  renderMachines(); // YOUR EXISTING UI
}


// ================= STREAM DATA (REPLACES FAKE DATA) =================
async function apiStream(){

  const id = state.selectedMachine;
  if(!id) return;

  const res = await fetch(`${BASE_URL}/stream/${id}`);
  const d = await res.json();

  state.readings[id] = {
    temp: d.temperature,
    vib: d.vibration,
    rpm: d.rpm,
    curr: d.current,
    status: d.status
  };

  updateChartWithNewPoint(id, d); // YOUR EXISTING FUNCTION
  renderMachines();
}


// ================= HISTORY =================
async function apiHistory(id){

  const res = await fetch(`${BASE_URL}/history/${id}`);
  const data = await res.json();

  state.history[id] = data;

  initChart(id); // YOUR EXISTING CHART FUNCTION
}


// ================= ALERTS =================
async function apiAlerts(){

  const res = await fetch(BASE_URL + "/alerts");
  const data = await res.json();

  state.alerts = data;

  renderAlerts(); // YOUR EXISTING UI
}


// ================= CREATE ALERT =================
async function apiCreateAlert(){

  await fetch(BASE_URL + "/alert",{
    method:"POST",
    headers:{ "Content-Type":"application/json" },
    body: JSON.stringify({
      machine_id: state.selectedMachine,
      type:"manual"
    })
  });

  apiAlerts();
}


// ================= SCHEDULE =================
async function apiSchedule(){

  await fetch(BASE_URL + "/schedule-maintenance",{
    method:"POST",
    headers:{ "Content-Type":"application/json" },
    body: JSON.stringify({
      machine_id: state.selectedMachine,
      priority:"high"
    })
  });

  alert("Maintenance Scheduled");
}


// ================= INIT APP =================
function startApp(){

  apiLoadMachines();
  apiAlerts();

  // load history for first machine

    if(state.selectedMachine){
      apiHistory(state.selectedMachine);
    }
  },500);

  // LIVE STREAM (replaces simulation)
  setInterval(apiStream, 2000);
}
